Подключение

Инициализация Socket.IO

import { io } from 'socket.io-client';

const socket = io('https://api.floodilka.com', {
  auth: {
    token: 'your-jwt-token'
  },
  transports: ['websocket', 'polling']
});

socket.on('connect', () => {
  console.log('Connected:', socket.id);
});

socket.on('disconnect', (reason) => {
  console.log('Disconnected:', reason);
});

socket.on('error', (error) => {
  console.error('Socket error:', error);
});

JWT токен передается при подключении. При истечении токена соединение будет разорвано с кодом AUTH_EXPIRED.

Сообщения в каналах

message:send Клиент → Сервер

Отправляет сообщение в канал.

socket.emit('message:send', {
  channelId: '507f1f77bcf86cd799439011',
  content: 'Hello, world!',
  attachments: [],  // опционально
  replyTo: '507f1f77bcf86cd799439012'  // опционально
});
message:new Сервер → Клиент

Новое сообщение в канале.

socket.on('message:new', (data) => {
  // data.message - объект Message
  // data.channelId - ID канала
});
message:edit Клиент → Сервер

Редактирование своего сообщения.

socket.emit('message:edit', {
  messageId: '507f1f77bcf86cd799439011',
  content: 'Updated content'
});
message:edited Сервер → Клиент

Сообщение было отредактировано.

socket.on('message:edited', (data) => {
  // data.messageId, data.content, data.editedAt
});
message:delete Клиент → Сервер

Удаление сообщения.

socket.emit('message:delete', {
  messageId: '507f1f77bcf86cd799439011'
});
message:deleted Сервер → Клиент

Сообщение было удалено.

socket.on('message:deleted', (data) => {
  // data.messageId, data.channelId
});

socket.emit('reaction:remove', {

socket.on('direct-message:new', (data) => {

Серверы и каналы

server:join Клиент → Сервер

Подписка на события сервера.

socket.emit('server:join', {
  serverId: '507f1f77bcf86cd799439011'
});
channel:join Клиент → Сервер

Вход в текстовый канал для получения сообщений.

socket.emit('channel:join', {
  channelId: '507f1f77bcf86cd799439011'
});
channel:created Сервер → Клиент

Создан новый канал на сервере.

channel:updated Сервер → Клиент

Настройки канала обновлены.

channel:deleted Сервер → Клиент

Канал удален.

server:member-joined Сервер → Клиент

Новый участник присоединился к серверу.

server:member-left Сервер → Клиент

Участник покинул сервер.

server:deleted Сервер → Клиент

Сервер удален.

Голосовые каналы

voice:join Клиент → Сервер

Вход в голосовой канал.

socket.emit('voice:join', {
  channelId: '507f1f77bcf86cd799439011',
  serverId: '507f1f77bcf86cd799439012'
});
voice:joined Сервер → Клиент

Успешно вошел в голосовой канал.

socket.on('voice:joined', (data) => {
  // data.channelId
  // data.routerRtpCapabilities - для mediasoup
  // data.users - список пользователей в канале
});
voice:leave Клиент → Сервер

Выход из голосового канала.

socket.emit('voice:leave', {
  channelId: '507f1f77bcf86cd799439011'
});
voice:user-joined Сервер → Клиент

Пользователь присоединился к голосовому каналу.

voice:user-left Сервер → Клиент

Пользователь покинул голосовой канал.

voice:mute / voice:unmute Клиент ↔ Сервер

Переключение микрофона.

voice:deafen / voice:undeafen Клиент ↔ Сервер

Переключение звука (глушение входящего аудио).

voice:produce Клиент → Сервер

Начало вещания аудио/видео (mediasoup).

socket.emit('voice:produce', {
  channelId: '507f1f77bcf86cd799439011',
  kind: 'audio',  // или 'video'
  rtpParameters: { ... }  // параметры mediasoup
});
voice:user-speaking Сервер → Клиент

Индикатор говорящего пользователя.

Звонки (1-на-1)

call:initiate Клиент → Сервер

Начать звонок.

socket.emit('call:initiate', {
  calleeId: '507f1f77bcf86cd799439011',
  type: 'audio'  // или 'video'
});
call:incoming Сервер → Клиент

Входящий звонок.

socket.on('call:incoming', (data) => {
  // data.callId
  // data.caller - информация о звонящем
  // data.type - 'audio' или 'video'
});
call:accept Клиент → Сервер

Принять звонок.

socket.emit('call:accept', {
  callId: '507f1f77bcf86cd799439011'
});
call:decline Клиент → Сервер

Отклонить звонок.

call:end Клиент → Сервер

Завершить звонок.

call:ended Сервер → Клиент

Звонок завершен.

socket.on('call:ended', (data) => {
  // data.callId
  // data.reason - причина завершения
  // data.duration - длительность в секундах
});
call:missed Сервер → Клиент

Пропущенный звонок.

socket.on('user:status-changed', (data) => {
  // data.userId
  // data.status
});

Коды ошибок

Код Описание
AUTH_REQUIREDТребуется аутентификация
AUTH_INVALIDНеверный токен
AUTH_EXPIREDТокен истек
PERMISSION_DENIEDНедостаточно прав
NOT_FOUNDРесурс не найден
RATE_LIMITEDСлишком много запросов
ALREADY_IN_CHANNELУже в голосовом канале
CALL_IN_PROGRESSУже идет звонок
USER_BUSYПользователь занят
USER_BLOCKEDПользователь заблокирован

Обработка ошибок

socket.on('error', (error) => {
  console.error('Socket error:', error.code, error.message);

  switch (error.code) {
    case 'AUTH_EXPIRED':
      // Переподключиться с новым токеном
      break;
    case 'RATE_LIMITED':
      // Подождать перед повтором
      break;
    default:
      // Показать сообщение пользователю
  }
});
